```python
# backend/src/digimon_world/world/narrator.py
"""NarratorSystem - 世界叙事引擎。每 N tick 收集重大事件 → LLM 写故事摘要。"""

class NarratorSystem:
    def __init__(self, interval: int = 100):
        self.journal: list[dict] = []
        self.narration_interval = interval
        self._tick_counter = 0
        self._last_narrated_at = 0

    def tick(self, world, timeline_system):
        self._tick_counter += 1
        if self._tick_counter - self._last_narrated_at < self.narration_interval:
            return
        self._last_narrated_at = self._tick_counter
        # 收集事件 (Task 2 实现完整逻辑)
        events = timeline_system.build(world, limit=30)
        if not events:
            return
        narrative = self._compose(world, events)
        if narrative:
            self.journal.append(narrative)

    def _compose(self, world, events):
        # Task 3 实现 LLM 调用
        return {"tick": self._tick_counter, "events": len(events), "story": "占位叙事"}

_narrator: NarratorSystem | None = None

def get_narrator() -> NarratorSystem:
    global _narrator
    if _narrator is None:
        _narrator = NarratorSystem()
    return _narrator
```

```python
def _collect_context(self, world, events):
    evolution_count = sum(1 for e in events if e["type"] == "evolution")
    battle_count = sum(1 for e in events if e["type"] == "battle")
    story_events = [e for e in events if e["type"] == "story_event"]
    first_meets = [e for e in events if e["type"] == "first_meet"]
    
    # 按重要性排序
    sorted_events = sorted(events, key=lambda e: e["importance"], reverse=True)
    
    return {
        "tick": world.tick if hasattr(world, 'tick') else 0,
        "agent_count": world.count(),
        "events": sorted_events[:20],
        "evolution_count": evolution_count,
        "battle_count": battle_count,
        "story_events": story_events,
        "first_meets": first_meets,
    }
```

```python
async def _compose_async(self, context):
    from ..llm.client import get_llm_client
    client = get_llm_client()
    
    # 构建 prompt: 把事件列表格式化
    event_lines = []
    for e in context["events"][:10]:
        event_lines.append(f"- {e['icon']} [{e['type']}] {e['title']}")
    
    prompt = f"""你是数码世界的说书人。根据以下事件，写一段 2-3 句话的世界故事摘要，像动画旁白一样生动。

世界状态: {context['agent_count']} 只数码兽
进化: {context['evolution_count']} 次, 战斗: {context['battle_count']} 场

今日大事:
{chr(10).join(event_lines)}

请用中文回复，格式:
标题: <15字以内的故事标题>
摘要: <2-3句叙事>"""
    
    response = await client.chat(prompt, system="你是数码世界的叙事者，用动画旁白风格讲述世界故事。")
    
    # 解析标题和摘要
    title = "数码世界的一天"
    story = response
    for line in response.split("\n"):
        if line.startswith("标题"):
            title = line.split(":", 1)[-1].strip()
        elif line.startswith("摘要"):
            story = line.split(":", 1)[-1].strip()
    
    return {
        "tick": context["tick"],
        "title": title,
        "story": story,
        "events_count": len(context["events"]),
        "evolution_count": context["evolution_count"],
        "battle_count": context["battle_count"],
    }
```

```python
def get_narratives(limit: int = 10) -> dict:
    n = get_narrator()
    journal = n.journal[-limit:]
    return {"count": n.narration_count, "entries": journal}

@app.get("/api/narratives/latest")
def get_latest_narrative() -> dict:
    n = get_narrator()
    if not n.journal:
        raise HTTPException(404, "No narratives yet")
    return n.journal[-1]
```

```javascript
async function fetchNarrative() {
  try {
    const res = await fetch(API_BASE + '/api/narratives/latest');
    if (!res.ok) return;
    const data = await res.json();
    document.querySelector('.narrative-title').textContent = data.title;
```
